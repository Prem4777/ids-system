# Simple DoS (same laptop)

# 1) Activate venv from project root
(Set-ExecutionPolicy -Scope Process -ExecutionPolicy RemoteSigned) ; (& ".\.venv\Scripts\Activate.ps1")

# 2) Start Phase 2 stack (backend + dashboard + profiler)
cmd /c .\scripts\start_phase2.bat

# 3) Run one simple DoS burst from project root
.\.venv\Scripts\python.exe .\simulation\burst_traffic.py --target 127.0.0.1 --port 8000 --rounds 1 --burst-seconds 10 --cooldown-seconds 1 --workers 25 --request-rate 120 --jitter-ms 3

# 4) Verify detection quickly
Invoke-RestMethod http://127.0.0.1:8000/events?limit=6 | ConvertTo-Json -Depth 6
